array of numbers

$numbers = array(4,1,7,11,-1,9,33,-4);
print_r($numbers);
hr();

echo 'Sorted by value: sort
';
sort($numbers);
print_r($numbers);

';

associative array

$person = array(
	'firstname'		=> 'Duncan',
	'lastname'		=> 'Heal',
	'gender'		=> 'male',
	'age'			=> 36,
	'occupation'	=> 'developer'
	);
print_r($person);
hr();

echo 'Sorted by key: ksort
';
ksort($person);
print_r($person);

echo 'Sorted by value: asort
';
asort($person);
print_r($person);





range

$range = range(1, 10);
print_r($range);
hr();

echo '1 to 10 mixed up: shuffle
';
shuffle($range);
print_r($range);

echo 'Random number from 1 to 20: array_rand (gives a random key, not the value)
';
$random_key = array_rand($range);
echo $range[$random_key];
